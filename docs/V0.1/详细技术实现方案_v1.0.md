### Task T6 [DeepSeek-TUI]：Linux MemoryCollector 实现

**文件**：`src/collector/linux.rs`

**执行指令**：
```rust
use anyhow::{Context, Result};
use crate::collector::MemoryCollector;
use crate::metrics::{SystemMetrics, PressureLevel};

pub struct LinuxCollector;

impl LinuxCollector {
    pub fn new() -> Self { Self }
}

impl MemoryCollector for LinuxCollector {
    fn collect(&self) -> Result<SystemMetrics> {
        let content = std::fs::read_to_string("/proc/meminfo")
            .context("Failed to read /proc/meminfo")?;

        let mut total_kb = 0u64;
        let mut available_kb = 0u64;
        let mut cached_kb = 0u64;
        let mut buffers_kb = 0u64;

        for line in content.lines() {
            if let Some((key, value)) = line.split_once(':') {
                let val = value.trim().split_whitespace().next()
                    .and_then(|v| v.parse::<u64>().ok())
                    .unwrap_or(0);
                match key {
                    "MemTotal" => total_kb = val,
                    "MemAvailable" => available_kb = val,
                    "Cached" => cached_kb = val,
                    "Buffers" => buffers_kb = val,
                    _ => {}
                }
            }
        }

        if total_kb == 0 {
            anyhow::bail!("Failed to parse MemTotal from /proc/meminfo");
        }

        // Fallback for kernels < 3.14 without MemAvailable
        if available_kb == 0 {
            available_kb = total_kb.saturating_sub(
                content.lines()
                    .filter_map(|l| l.split_once(':'))
                    .filter(|(k, _)| matches!(*k, "MemFree" | "Buffers" | "Cached" | "SReclaimable"))
                    .filter_map(|(_, v)| v.trim().split_whitespace().next()?.parse::<u64>().ok())
                    .sum::<u64>()
            );
            // Actually simpler: use MemFree + Cached + Buffers estimate
            // For now, just note the limitation
        }

        let total_mb = total_kb / 1024;
        let available_mb = available_kb / 1024;
        let reclaimable_mb = (cached_kb + buffers_kb) / 1024;
        let used_percent = ((total_mb - available_mb) as f64 / total_mb as f64 * 100.0);
        let used_percent = (used_percent * 10.0).round() / 10.0;
        let pressure = determine_pressure(available_mb, used_percent);

        Ok(SystemMetrics {
            total_mb,
            available_mb,
            used_percent,
            reclaimable_mb,
            pressure,
        })
    }
}

fn determine_pressure(available_mb: u64, used_percent: f64) -> PressureLevel {
    if available_mb < 2048 || used_percent > 92.0 {
        PressureLevel::Critical
    } else if available_mb < 4096 || used_percent > 80.0 {
        PressureLevel::High
    } else if available_mb < 8192 || used_percent > 50.0 {
        PressureLevel::Medium
    } else {
        PressureLevel::Low
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_pressure_critical() {
        assert_eq!(determine_pressure(1500, 50.0), PressureLevel::Critical);
        assert_eq!(determine_pressure(8192, 95.0), PressureLevel::Critical);
    }

    #[test]
    fn test_pressure_high() {
        assert_eq!(determine_pressure(3000, 50.0), PressureLevel::High);
        assert_eq!(determine_pressure(8192, 85.0), PressureLevel::High);
    }
}
```

### Task T7 [DeepSeek-TUI]：macOS MemoryCollector 实现

**文件**：`src/collector/macos.rs`

**执行指令**：
```rust
use anyhow::{Context, Result};
use crate::collector::MemoryCollector;
use crate::metrics::{SystemMetrics, PressureLevel};

pub struct MacOSCollector;

impl MacOSCollector {
    pub fn new() -> Self { Self }
}

impl MemoryCollector for MacOSCollector {
    fn collect(&self) -> Result<SystemMetrics> {
        let total_bytes = sysctl_val("hw.memsize")
            .context("Failed to get hw.memsize")?;
        let total_mb = (total_bytes / (1024 * 1024)) as u64;

        let page_size = sysctl_val("vm.pagesize")
            .context("Failed to get vm.pagesize")?;
        let free_pages = sysctl_val("vm.vm_page_free_count")
            .context("Failed to get free pages")?;
        let inactive_pages = sysctl_val("vm.vm_page_inactive_count")
            .context("Failed to get inactive pages")?;
        let speculative_pages = sysctl_val("vm.vm_page_speculative_count")
            .context("Failed to get speculative pages")?;
        let purgeable_pages = sysctl_val("vm.vm_page_purgeable_count")
            .context("Failed to get purgeable pages")?;

        let available_bytes = (free_pages + inactive_pages + speculative_pages + purgeable_pages / 2) * page_size;
        let available_mb = (available_bytes / (1024 * 1024)) as u64;

        let pressure_level = get_memory_pressure();
        let used_percent = ((total_mb - available_mb) as f64 / total_mb as f64 * 100.0);
        let used_percent = (used_percent * 10.0).round() / 10.0;
        let reclaimable_mb = ((inactive_pages + speculative_pages) * page_size / (1024 * 1024)) as u64;

        let pressure = match pressure_level {
            Some(4) => PressureLevel::Critical,
            Some(3) => PressureLevel::High,
            Some(2) => PressureLevel::Medium,
            Some(1) => PressureLevel::Low,
            _ => determine_pressure_fallback(available_mb, used_percent),
        };

        Ok(SystemMetrics {
            total_mb,
            available_mb,
            used_percent,
            reclaimable_mb,
            pressure,
        })
    }
}

fn sysctl_val(name: &str) -> Result<u64> {
    let c_name = std::ffi::CString::new(name)
        .map_err(|e| anyhow::anyhow!("Invalid sysctl name {}: {}", name, e))?;
    let mut val: u64 = 0;
    let mut len = std::mem::size_of::<u64>();
    let ret = unsafe {
        libc::sysctlbyname(
            c_name.as_ptr(),
            &mut val as *mut _ as *mut libc::c_void,
            &mut len,
            std::ptr::null_mut(),
            0,
        )
    };
    if ret != 0 {
        anyhow::bail!("sysctlbyname {} failed with code {}", name, ret);
    }
    Ok(val)
}

fn get_memory_pressure() -> Option<u64> {
    let c_name = std::ffi::CString::new("kern.memorystatus_vm_pressure_level").ok()?;
    let mut val: u64 = 0;
    let mut len = std::mem::size_of::<u64>();
    let ret = unsafe {
        libc::sysctlbyname(
            c_name.as_ptr(),
            &mut val as *mut _ as *mut libc::c_void,
            &mut len,
            std::ptr::null_mut(),
            0,
        )
    };
    if ret != 0 { None } else { Some(val) }
}

fn determine_pressure_fallback(available_mb: u64, used_percent: f64) -> PressureLevel {
    if available_mb < 2048 || used_percent > 92.0 {
        PressureLevel::Critical
    } else if available_mb < 4096 || used_percent > 80.0 {
        PressureLevel::High
    } else if available_mb < 8192 || used_percent > 50.0 {
        PressureLevel::Medium
    } else {
        PressureLevel::Low
    }
}
```
