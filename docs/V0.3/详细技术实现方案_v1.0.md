// 模型名哈希：SHA256 → 前 16 位 hex（CR-07 安全专家要求 ≥ 16 位）
pub fn hash_model_name(name: &str) -> String {
    use sha2::{Sha256, Digest};
    let hash = Sha256::digest(name.as_bytes());
    hex::encode(&hash[..8]) // 前 8 字节 = 16 hex 字符
}

#[cfg(test)]

// 文件路径：~/.config/hawk-eye-mem/calibration.csv

    /// 确保父目录存在
    fn ensure_dir(&self) -> Result<()> {

            std::fs::create_dir_all(parent)

        }
        Ok(())

        file.try_lock_exclusive().is_ok()

use crate::metrics::ResourceSnapshot;

pub struct CalibrationEngine {
    store: Box<dyn CalibrationStore>,
    model_name: Option<String>,
}

impl CalibrationEngine {
    pub fn new(store: Box<dyn CalibrationStore>, model_name: Option<String>) -> Self {
        Self { store, model_name }
    }
    
    /// 记录推理前后的内存变化（CR-01：MCP Tool 传 tokens_processed）
    pub fn record_inference(
        &self,
        before: &ResourceSnapshot,
        after: &ResourceSnapshot,
        tokens_processed: u64,
    ) -> Result<Option<CalibrationPoint>> {
        let model = match &self.model_name {
            Some(m) => m.clone(),
            None => return Ok(None), // 未指定模型名，不校准
        };
        
        if tokens_processed == 0 { return Ok(None); }
        
        let delta_mb = before.memory.available_mb
            .saturating_sub(after.memory.available_mb);
        
        // 过滤噪声：内存变化 < 10MB 不记录
        if delta_mb < 10 { return Ok(None); }
        
        let bytes_per_token = (delta_mb as u64) * 1024 * 1024 / tokens_processed;
        let point = CalibrationPoint {
            timestamp: std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap()
                .as_secs(),
            bytes_per_token,
            tokens_processed,
        };
        
        self.store.append(point.clone(), &model)?;
        Ok(Some(point))
    }
    
    /// 获取修正后的参数（加权平均 + confidence 判定）
    pub fn get_corrected_params(&self) -> Result<Option<CorrectedParams>> {
        let model = match &self.model_name {
            Some(m) => m,
            None => return Ok(None),
        };
        
        let model_hash = hash_model_name(model);
        let points = self.store.read_by_model(&model_hash)?;
        if points.len() < 10 { return Ok(None); } // 样本不够，保守模式
        
        // 详见 T9 算法实现
        Ok(Some(CorrectedParams { /* ... */ }))
    }
}

pub struct CorrectedParams {

    pub safety_margin: f64,  // 校准后降至 10%

    pub calibration: CalibrationMeta,
}

// StateMachine：基于时间窗口 + 最少次数的压力状态转换（CR-02 架构师要求）

#[derive(Debug, Clone, PartialEq)]
pub enum MonitorState {
    Normal,
    Warning,
    Critical,
}

pub struct StateMachine {
    state: MonitorState,
    state_entered_at: Instant,
    consecutive_high: u32,      // 连续高压计数
    consecutive_critical: u32,  // 连续 critical 计数
    consecutive_ok: u32,        // 连续正常计数
    config: StateMachineConfig,
}

pub struct StateMachineConfig {
    pub warning_seconds: u64,       // 默认 10
    pub critical_seconds: u64,      // 默认 15
    pub recovery_seconds: u64,      // 默认 10
    pub min_samples_warning: u32,   // CR-02：最少采集次数，默认 3
    pub min_samples_critical: u32,  // 默认 5
    pub min_samples_recovery: u32,  // 默认 3
    pub emergency_available_mb: u64,  // CR-08：紧急阈值，默认 512
    pub emergency_used_percent: f64,  // CR-08：紧急阈值，默认 98.0
}

impl StateMachine {
    pub fn new(config: StateMachineConfig) -> Self { /* ... */ }
    
    pub fn update(&mut self, pressure: &PressureLevel, now: Instant) -> Option<StateTransition> {
        // 1. CR-08 紧急快速通道：available < 512MB 或 used > 98%，立即 Critical
        if self.is_emergency(pressure) {
            return Some(StateTransition::Emergency);
        }
        
        // 2. 正常状态转换逻辑
        let duration = now.duration_since(self.state_entered_at);
        match (&self.state, pressure) {
            // Normal → Warning：持续 10s 且至少 3 次 high/critical
            (Normal, High | Critical) => {
                self.consecutive_high += 1;
                if duration >= Duration::from_secs(config.warning_seconds)
                    && self.consecutive_high >= config.min_samples_warning {
                    self.transition_to(Warning, now)
                } else { None }
            }
            // Warning → Critical：持续 15s 且至少 5 次 critical
            (Warning, Critical) => {
                self.consecutive_critical += 1;
                if duration >= Duration::from_secs(config.critical_seconds)
                    && self.consecutive_critical >= config.min_samples_critical {
                    Some(CriticalTransition)
                } else { None }
            }
            // Warning → Normal：持续 10s 且至少 3 次 ≤ medium
            (Warning, Low | Medium) => {
                self.consecutive_ok += 1;
                if duration >= Duration::from_secs(config.recovery_seconds)
                    && self.consecutive_ok >= config.min_samples_recovery {
                    Some(NormalTransition)
                } else { None }
            }
            // Critical → Warning 同理...
            _ => None
        }
    }
    
    /// CR-08：紧急快速通道
    fn is_emergency(&self, metrics: &SystemMetrics) -> bool {
        metrics.available_mb < self.config.emergency_available_mb
            || metrics.used_percent > self.config.emergency_used_percent
    }
}
```

// 加权平均计算：最近 10 次线性衰减权重
// 最新记录权重 1.0，第 10 条权重 0.1
// 样本数 ≥ 10 且变异系数 ≤ 20% → calibrated
// 安全边际从 30% 降至 10%

pub struct CalibrationAlgorithm;

impl CalibrationAlgorithm {
    pub fn compute(points: &[CalibrationPoint]) -> Option<CorrectedParams> {
        if points.len() < 10 { return None; }
        
        let recent: Vec<_> = points.iter().rev().take(10).collect();
        let weights: Vec<f64> = (0..recent.len())
            .map(|i| 1.0 - i as f64 * 0.1)  // 1.0, 0.9, 0.8, ..., 0.1
            .collect();
        
        let total_weight: f64 = weights.iter().sum();
        let avg: f64 = recent.iter().zip(&weights)
            .map(|(p, w)| p.bytes_per_token as f64 * w)
            .sum::<f64>() / total_weight;
        
        let variance: f64 = recent.iter().zip(&weights)
            .map(|(p, w)| w * (p.bytes_per_token as f64 - avg).powi(2))
            .sum::<f64>() / total_weight;
        let std_dev = variance.sqrt();
        let cv = std_dev / avg; // 变异系数
        
        let confidence = if cv <= 0.20 {
            Confidence::Calibrated
        } else {
            Confidence::Conservative
        };
        
        let safety_margin = if confidence == Confidence::Calibrated {
            10.0  // 校准后降至 10%
        } else {
            30.0  // 回退保守
        };
        
        // 趋势判定
        let trend = if recent.len() >= 4 {
            let first_half: f64 = recent[recent.len()/2..].iter()
                .map(|p| p.bytes_per_token as f64).sum::<f64>() / (recent.len()/2) as f64;
            let second_half: f64 = recent[..recent.len()/2].iter()
                .map(|p| p.bytes_per_token as f64).sum::<f64>() / (recent.len()/2) as f64;
            if (first_half - second_half).abs() < 50.0 { "stable".to_string() }
            else if second_half < first_half { "improving".to_string() }
            else { "degrading".to_string() }
        } else {
            "unknown".to_string()
        };
        
        Some(CorrectedParams {
            avg_bytes_per_token: avg.round() as u64,
            safety_margin,
            confidence,
            calibration: CalibrationMeta {
                samples: points.len(),
                avg_bytes_per_token: avg.round() as u64,
                std_dev: (std_dev * 10.0).round() / 10.0,
                trend,
            },
        })
    }
}
```

**降级方案**：NVML 初始化失败时自动降级到 nvidia-smi 解析（V0.2 方案），stderr 输出 warning。

**约束**：

- 编译条件：`#[cfg(feature = "gpu")]`
